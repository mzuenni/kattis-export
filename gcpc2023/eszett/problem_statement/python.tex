\documentclass[varwidth=6cm]{standalone}
\usepackage{minted}

\begin{minted}{python}
a = input().lower()
if a.find('sss') != -1:
  print(a.replace('sss', 'sB'))
  print(a.replace('sss', 'Bs'))
elif a.find('ss') != -1:
  print(a.replace('ss', 'B'))
print(a)
\end{minted}
